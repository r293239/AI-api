# Quick Start Example from ai_learning_system import LearningAI, GridWorldEnvironment # Create the learning AI ai = LearningAI() # Add a navigation task grid_env = GridWorldEnvironment(width=5, height=5) ai.add_task("navigation", grid_env) # Train the AI ai.learn_task("navigation", episodes=500) # Test the trained AI ai.demonstrate_task("navigation", episodes=3) # Get learning progress progress = ai.get_learning_progress("navigation") print(f"Best reward: {progress['best_reward']}")
